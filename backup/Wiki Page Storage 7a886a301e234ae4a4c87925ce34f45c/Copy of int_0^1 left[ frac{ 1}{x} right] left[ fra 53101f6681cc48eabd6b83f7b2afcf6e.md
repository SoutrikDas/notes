& \int_0^1 \left[ \frac{ 1}{x}\right] \left[ \frac{1}{1-x} \right] \left[ 1-\frac{1}{x}\right]dx  \\
& =\int_0^1 \left[ \frac{ 1}{1-x}\right] \left[ \frac{1}{x} \right] \left[ \frac{x}{x-1}\right]dx 
\end{aligned}$$
